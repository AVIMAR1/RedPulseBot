        function animateBalanceElement(element, newValue, oldValue) {
            if (!element) return;
            const formattedValue = newValue.toLocaleString();

            element.classList.remove('updating', 'increased');

            element.classList.add('updating');
            if (newValue > oldValue && oldValue > 0) {

        function getRandomSlotSymbol() {
            const total = slotsWeights.reduce((a, b) => a + b, 0);
            let random = Math.random() * total;
            for (let i = 0; i < slotsSymbols.length; i++) {
                if (random < slotsWeights[i]) return slotsSymbols[i];
                random -= slotsWeights[i];
            }
            return slotsSymbols[slotsSymbols.length - 1];
        }

            if (casinoState.slotsSpinning) return;

            const bet = parseInt(document.getElementById('slotsBet').value) || 0;

            updateUI();

                document.getElementById('slotsResult').innerHTML = `<p style="color:#27ae60;">СЂСџР‹В° Р СџР С•Р В±Р ВµР Т‘Р В°! x${multiplier} = +${win} СЂСџР„в„ў</p>`;
